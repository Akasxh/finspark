% ─── Problem Statement ───────────────────────────────────────────────────────
\begin{frame}{The Problem}
\begin{columns}
\begin{column}{0.55\textwidth}
\textbf{Indian lending platforms} integrate with 8+ external APIs:
\begin{itemize}
    \item CIBIL/Experian credit bureaus
    \item Aadhaar eKYC verification
    \item GST verification service
    \item Payment gateways (Razorpay)
    \item Fraud detection engines
    \item SMS/Email notification
    \item Account Aggregator (AA)
\end{itemize}
\end{column}
\begin{column}{0.4\textwidth}
\alert{Pain points:}
\begin{itemize}
    \item Manual configuration takes \textbf{2--4 weeks}
    \item Field mapping errors cause \textbf{data loss}
    \item No standardized testing before deployment
    \item Each adapter has different auth, schema, SLA
\end{itemize}
\end{column}
\end{columns}
\end{frame}

% ─── Solution ────────────────────────────────────────────────────────────────
\begin{frame}{AdaptConfig Solution}
\begin{center}
\textbf{Upload a BRD/API spec} $\rightarrow$ \textbf{AI generates config} $\rightarrow$ \textbf{Simulate \& validate} $\rightarrow$ \textbf{Deploy}
\end{center}
\vspace{0.5em}
\begin{columns}
\begin{column}{0.48\textwidth}
\textbf{Core Capabilities:}
\begin{enumerate}
    \item Document parsing (PDF, DOCX, YAML, JSON)
    \item AI-powered field mapping (Gemini 3)
    \item Rule-based validation \& augmentation
    \item Mock API simulation (8 adapters)
    \item Configuration lifecycle management
    \item Multi-tenant audit trail
\end{enumerate}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Tech Stack:}
\begin{itemize}
    \item \textbf{Backend}: FastAPI + SQLAlchemy + asyncio
    \item \textbf{Frontend}: React 18 + TypeScript + Tailwind
    \item \textbf{AI}: Google Gemini 3 Flash (REST)
    \item \textbf{DB}: SQLite (dev) / PostgreSQL (prod)
    \item \textbf{Testing}: 800+ tests, 82\% coverage
    \item \textbf{Deploy}: Docker Compose
\end{itemize}
\end{column}
\end{columns}
\end{frame}

% ─── Security ────────────────────────────────────────────────────────────────
\begin{frame}{Security \& Compliance}
\begin{columns}
\begin{column}{0.48\textwidth}
\textbf{Authentication \& Authorization:}
\begin{itemize}
    \item JWT Bearer token auth (production)
    \item Role-based access control (RBAC)
    \item Multi-tenant isolation
    \item Secret key validation at startup
\end{itemize}
\vspace{0.5em}
\textbf{Data Protection:}
\begin{itemize}
    \item PII masking (Aadhaar, PAN, phone)
    \item Fernet AES-256 encryption for secrets
    \item Path traversal protection on uploads
    \item File size \& type validation
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Infrastructure:}
\begin{itemize}
    \item CORS restricted to known origins
    \item Security headers (CSP, HSTS, X-Frame)
    \item Rate limiting (100 req/min)
    \item Structured logging (JSON)
    \item SSRF protection on webhooks
\end{itemize}
\vspace{0.5em}
\textbf{Testing:}
\begin{itemize}
    \item 800+ automated tests
    \item 82\% code coverage
    \item CI/CD via GitHub Actions
    \item Docker Compose deployment
\end{itemize}
\end{column}
\end{columns}
